<script>

let questions = [
["2 + 2?", ["4","3"], "4"],
["Capital of Japan?", ["Tokyo","Seoul"], "Tokyo"],
["Sun rises in?", ["East","West"], "East"],
["5 x 2?", ["10","8"], "10"],
["Sky color?", ["Blue","Green"], "Blue"],
["Earth is?", ["Planet","Star"], "Planet"],
["10 / 2?", ["5","6"], "5"],
["HTML stands for?", ["HyperText","HomeTool"], "HyperText"],
["Water freezes at?", ["0","100"], "0"],
["Opposite of hot?", ["Cold","Warm"], "Cold"]
];

let index = 0;
let xp = 0;
let level = 1;
let streak = 0;
let time = 300;

function addMsg(text, type){
    let div = document.createElement("div");
    div.className = "msg " + type;
    div.innerText = text;
    document.getElementById("chatbox").appendChild(div);
    chatbox.scrollTop = chatbox.scrollHeight;
}

function showQuestion(){
    let q = questions[index];
    addMsg("🤖 " + q[0], "bot");

    let choicesDiv = document.getElementById("choices");
    choicesDiv.innerHTML = "";

    q[1].forEach(choice=>{
        let btn = document.createElement("button");
        btn.innerText = choice;
        btn.onclick = ()=> answer(choice);
        choicesDiv.appendChild(btn);
    });
}

function answer(choice){

    let q = questions[index];

    addMsg("🧑 " + choice, "user");

    if(choice === q[2]){
        addMsg("✅ Correct! Nice!", "bot");
        xp += 10 + (streak * 2);
        streak++;
    } else {
        addMsg("❌ Oops! Correct was " + q[2], "bot");
        streak = 0;
    }

    if(xp >= level * 50){
        level++;
        addMsg("🎉 Level Up! Now level " + level, "bot");
    }

    updateStats();

    index++;

    if(index < questions.length){
        setTimeout(showQuestion, 1000);
    } else {
        endQuiz();
    }
}

function updateStats(){
    document.getElementById("xp").innerText = xp;
    document.getElementById("level").innerText = level;
    document.getElementById("streak").innerText = streak;
}

function endQuiz(){
    addMsg("🏁 Quiz finished!", "bot");
    addMsg("Your XP: " + xp, "bot");

    document.getElementById("choices").innerHTML = "";
}

// TIMER
setInterval(()=>{
    time--;
    document.getElementById("timer").innerText = time;

    if(time <= 0){
        addMsg("⏳ Time up!", "bot");
        endQuiz();
    }
},1000);

// START
addMsg("🤖 Hey Raiyan! Ready for a fun quiz?", "bot");
setTimeout(showQuestion, 1000);

</script>
